using System;

public class Funcionarios
{
    public string matricula;
    public string nome;
    public int contador = 0;

    public Funcionarios() { }

    public Funcionarios(string nome, string matricula)
    {
        this.nome = nome;
        this.matricula = matricula;
        AdicionarFuncionario();
    }

    public void ExibirDadosFuncionario()
    {
        Console.WriteLine("Dados do Funcionario");
        Console.WriteLine("Nome: " + this.nome);
        Console.WriteLine("Matricula: " + this.matricula);
        Console.WriteLine("Total de Funcionarios: " + ++this.contador);
    }

    public void AdicionarFuncionario()
    {
        Console.WriteLine("Adicionado o Funcionario " + ++this.contador);
    }
}
